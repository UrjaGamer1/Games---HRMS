    <h2>Match Column A with Column B</h2>
    <p><strong>Instructions:</strong> Drag the terms from Column A and drop them onto the correct descriptions in Column B. Once all pairs are matched, click 'Submit' to check your answers. Correct matches will turn green, and incorrect ones will turn red.</p>
    <div class="container">
        <svg id="lines"></svg>
        <div class="column" id="columnA">
            <h3>Column A</h3>
            <div class="item" draggable="true" id="item1">Legal Entity</div>
            <div class="item" draggable="true" id="item2">Locations</div>
            <div class="item" draggable="true" id="item3">Business Unit</div>
            <div class="item" draggable="true" id="item4">Pay Grades</div>
            <div class="item" draggable="true" id="item5">Departments</div>
        </div>
        <div class="column" id="columnB">
            <h3>Column B</h3>
            <script>
                let answers = [
                    {text: "Registered name of the business that hires employees and follows tax and labor laws.", id: "item1"},
                    {text: "Physical or geographical offices or sites", id: "item2"},
                    {text: "A specific operational division within an organization", id: "item3"},
                    {text: "A structured hierarchy defining salary ranges", id: "item4"},
                    {text: "Different sections handling specific organizational functions", id: "item5"}
                ];
                answers.sort(() => Math.random() - 0.5);
                document.write(answers.map(a => `<div class='dropzone' data-answer='${a.id}'>${a.text}</div>`).join(''));
            </script>
        </div>
    </div>
    <button onclick="checkAnswers()">Submit</button>
    <script>
        document.querySelectorAll('.item').forEach(item => {
            item.addEventListener('dragstart', (e) => {
                e.dataTransfer.setData('text', e.target.id);
            });
        });

        document.querySelectorAll('.dropzone').forEach(zone => {
            zone.addEventListener('dragover', (e) => {
                e.preventDefault();
            });
            zone.addEventListener('drop', (e) => {
                e.preventDefault();
                let data = e.dataTransfer.getData('text');
                let draggedElement = document.getElementById(data);
                e.target.innerHTML = draggedElement.outerHTML;
                e.target.dataset.matched = data;
            });
        });

        function checkAnswers() {
            document.querySelectorAll('.dropzone').forEach(zone => {
                if (zone.dataset.matched === zone.dataset.answer) {
                    zone.classList.add('correct');
                } else {
                    zone.classList.add('incorrect');
                }
            });
        }
    </script>
